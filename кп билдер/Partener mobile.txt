function downloadPDF(){
  if(!g('cl_n')&&!g('cl_co')){toast('❌ Заполни клиента');setPage('p-client');return;}
  var num=g('kp-num')||document.getElementById('hdr-num').textContent;
  var clientName=g('cl_n')||g('cl_co')||'Client';
  showLoading('Генерируем PDF...');
  var htmlContent=document.getElementById('pv-wrap').innerHTML;
  google.script.run.withSuccessHandler(function(res){
    hideLoading();
    if(res&&res.success){
      // Открываем PDF в новой вкладке — пользователь жмёт "Скачать" в просмотрщике
      window.open(res.url,'_blank');
      toast('✅ PDF готов');
    } else {
      toast('❌ '+(res&&res.error||'Ошибка генерации'));
    }
  }).withFailureHandler(function(e){hideLoading();toast('❌ '+e.message);})
   .generatePDFFromHtml(htmlContent,num,clientName);
}

function shareEmail(){
  var num=g('kp-num')||document.getElementById('hdr-num').textContent;
  var subject=encodeURIComponent('Коммерческое предложение QGROUP '+num);
  var body=encodeURIComponent(buildShareMsg());
  var to=g('cl_em')||'';
  window.location.href='mailto:'+to+'?subject='+subject+'&body='+body;
}

function shareCopy(){
  var msg=buildShareMsg();
  if(navigator.clipboard&&navigator.clipboard.writeText){
    navigator.clipboard.writeText(msg).then(function(){toast('📋 Скопировано');},function(){fallbackCopy(msg);});
  } else fallbackCopy(msg);
}
function fallbackCopy(text){
  var ta=document.createElement('textarea');ta.value=text;ta.style.position='fixed';ta.style.left='-9999px';document.body.appendChild(ta);ta.select();
  try{document.execCommand('copy');toast('📋 Скопировано');}catch(e){toast('❌ Не удалось');}
  document.body.removeChild(ta);
}
